void setup() {
  pinMode(11, OUTPUT); // Initializing GPIO pin 11 to light up LED
  pinMode(12, OUTPUT); // Initializing GPIO pin 12 to light up LED
  pinMode(13, OUTPUT); // Initializing GPIO pin 13 to light up LED
}

void loop() {

  // This is for my first red LED
  digitalWrite(11, HIGH); // Turn LED on
  delay(); //LED on for 500 ms
  digitalWrite(11, LOW); // Turn LED off
  delay(500); //LED off for 500 ms

  // This is for my green LED
  digitalWrite(12, HIGH); // Turn LED on
  delay(500); //LED on for 500 ms
  digitalWrite(12, LOW); // Turn LED off
  delay(500); //LED off for 500 ms

  //This is for my last red LED
  digitalWrite(13, HIGH); // Turn LED on
  delay(500); //LED on for 500 ms
  digitalWrite(13, LOW); // Turn LED off
  delay(500); //LED off for 500 ms
}
